#  Query Description : 
#  Retrieve a list of N creative works, their thumbnail and the thumbnail’s alternative text.
#  reasoning : rdfs:subClassOf, owl:propertyChainAxiom, owl:DataProperty 

SELECT DISTINCT ?thing ?thumbnail ?altText FROM <http://ldbc50M>
WHERE 
{ 
  ?thing rdf:type cwork:CreativeWork option(transitive). 
  ?thing cwork:thumbnail ?thumbnail . 
  OPTIONAL {
  	?thumbnail cwork:altText ?altText . 
	  ?thing ldbc:cworkThumbnailAltText ?altText2 . 
	  FILTER (?altText = ?altText2)  . 
  }
}
LIMIT 100
